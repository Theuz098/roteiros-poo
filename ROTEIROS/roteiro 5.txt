using System;

// Exercicio 1
class Exercicio1
{
    static void Main()
    {
        Console.WriteLine("Digite um número decimal:");
        decimal numero = decimal.Parse(Console.ReadLine());

        Console.WriteLine($"Valor padrão: {numero}");

        Console.WriteLine("Valor com arredondamento padrão: " +
                          Math.Round(numero));

        Console.WriteLine("Valor com arredondamento para baixo: " +
                          Math.Floor(numero));

        Console.WriteLine("Valor com arredondamento para cima: " +
                          Math.Ceiling(numero));

        Console.WriteLine("Valor com truncamento: " +
                          Math.Truncate(numero));
    }
}
