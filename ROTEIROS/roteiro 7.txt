using System;
using System.Collections.Generic;

class Exercicio1
{
    static void Main()
    {
        List<string> lista = new List<string>();
        int opcao = 1;

        while (opcao != 0)
        {
            Console.WriteLine("\n1 - Adicionar item");
            Console.WriteLine("2 - Remover item");
            Console.WriteLine("3 - Listar itens");
            Console.WriteLine("4 - Mostrar quantidade");
            Console.WriteLine("0 - Sair");
            Console.Write("Opção: ");
            int.TryParse(Console.ReadLine(), out opcao);

            if (opcao == 1)
            {
                Console.Write("Item: ");
                string item = Console.ReadLine();
                lista.Add(item);
            }
            else if (opcao == 2)
            {
                Console.Write("Item para remover: ");
                string item = Console.ReadLine();
                lista.Remove(item);
            }
            else if (opcao == 3)
            {
                Console.WriteLine("\nLista de compras:");
                foreach (var i in lista)
                {
                    Console.WriteLine("- " + i);
                }
            }
            else if (opcao == 4)
            {
                Console.WriteLine("Quantidade de itens: " + lista.Count);
            }
        }
    }
}

using System;
using System.Collections.Generic;

class Exercicio2
{
    static void Main()
    {
        Dictionary<int, string> produtos = new Dictionary<int, string>();
        int opcao = 1;

        while (opcao != 0)
        {
            Console.WriteLine("\n1 - Adicionar produto");
            Console.WriteLine("2 - Buscar produto por código");
            Console.WriteLine("3 - Listar todos os produtos");
            Console.WriteLine("0 - Sair");
            Console.Write("Opção: ");
            int.TryParse(Console.ReadLine(), out opcao);

            if (opcao == 1)
            {
                Console.Write("Código: ");
                int codigo = int.Parse(Console.ReadLine());

                Console.Write("Nome do produto: ");
                string nome = Console.ReadLine();

                produtos[codigo] = nome;
            }
            else if (opcao == 2)
            {
                Console.Write("Código para busca: ");
                int codigo = int.Parse(Console.ReadLine());

                if (produtos.ContainsKey(codigo))
                {
                    Console.WriteLine("Produto: " + produtos[codigo]);
                }
                else
                {
                    Console.WriteLine("Produto não encontrado.");
                }
            }
            else if (opcao == 3)
            {
                Console.WriteLine("\nProdutos cadastrados:");
                foreach (var p in produtos)
                {
                    Console.WriteLine("Código " + p.Key + " - " + p.Value);
                }
            }
        }
    }
}

3- using System;
using System.Collections.Generic;

class Exercicio3
{
    static void Main()
    {
        HashSet<int> numeros = new HashSet<int>();
        int n = -1;

        Console.WriteLine("Digite números inteiros (0 para parar):");

        while (n != 0)
        {
            Console.Write("Número: ");
            int.TryParse(Console.ReadLine(), out n);

            if (n != 0)
            {
                if (numeros.Add(n))
                {
                    Console.WriteLine("Número adicionado.");
                }
                else
                {
                    Console.WriteLine("Número repetido.");
                }
            }
        }

        Console.WriteLine("\nNúmeros únicos digitados:");
        foreach (int num in numeros)
        {
            Console.WriteLine(num);
        }
    }
}

4-using System;
using System.Collections.Generic;

class Exercicio4
{
    static void Main()
    {
        Queue<string> fila = new Queue<string>();
        int opcao = 1;

        while (opcao != 0)
        {
            Console.WriteLine("\n1 - Adicionar cliente na fila");
            Console.WriteLine("2 - Atender cliente");
            Console.WriteLine("3 - Mostrar quantidade na fila");
            Console.WriteLine("0 - Sair");
            Console.Write("Opção: ");
            int.TryParse(Console.ReadLine(), out opcao);

            if (opcao == 1)
            {
                Console.Write("Nome do cliente: ");
                string nome = Console.ReadLine();
                fila.Enqueue(nome);
            }
            else if (opcao == 2)
            {
                if (fila.Count > 0)
                {
                    string atendido = fila.Dequeue();
                    Console.WriteLine("Atendendo: " + atendido);
                }
                else
                {
                    Console.WriteLine("Fila vazia.");
                }
            }
            else if (opcao == 3)
            {
                Console.WriteLine("Quantidade na fila: " + fila.Count);
            }
        }
    }
}

5- using System;
using System.Collections.Generic;

class Exercicio5
{
    static void Main()
    {
        Console.Write("Digite uma palavra: ");
        string palavra = Console.ReadLine();

        Stack<char> pilha = new Stack<char>();

        foreach (char c in palavra)
        {
            pilha.Push(c);
        }

        Console.WriteLine("Palavra original: " + palavra);
        Console.Write("Palavra invertida: ");

        while (pilha.Count > 0)
        {
            Console.Write(pilha.Pop());
        }

        Console.WriteLine();
    }
}
